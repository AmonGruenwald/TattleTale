154{
155 // d - default, a - active, p - passive, n - name, o - detailed
156 char presentation = 'd';
157
158 constexpr auto parse(format_parse_context &ctx) -> decltype(ctx.begin())
159 {
160 auto it = ctx.begin(), end = ctx.end();
161 if (it != end && (*it == 'a' || *it == 'p' || *it == 'n' || *it == 'd' || *it == 'o'))
162 presentation = *it++;
163
164 if (it != end && *it != '}')
165 throw format_error("invalid format");
166
167 return it;
168 }
169
170 template <typename FormatContext>
171 auto format(tattletale::Kernel &kernel, FormatContext &ctx) -> decltype(ctx.out())
172 {
173 if (presentation == 'a')
174 {
175 return fmt::formatter<std::string>::format(kernel.GetActiveDescription(), ctx);
176 }
177 else if (presentation == 'p')
178 {
179 return fmt::formatter<std::string>::format(kernel.GetPassiveDescription(), ctx);
180 }
181 else if (presentation == 'n')

183 return fmt::formatter<std::string>::format(kernel.name_, ctx);
184 }
185 else if (presentation == 'o')
186 {
187 return fmt::formatter<std::string>::format(kernel.GetDetailedDescription(), ctx);
188 }

190 }
