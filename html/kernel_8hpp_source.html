178struct fmt::formatter<T, std::enable_if_t<std::is_base_of<tattletale::Kernel, T>::value, char>> : fmt::formatter<std::string>
179{
180 // d - default, a - active, p - passive, n - name, o - detailed
181 char presentation = 'd';
182
183 constexpr auto parse(format_parse_context &ctx) -> decltype(ctx.begin())
184 {
185 auto it = ctx.begin(), end = ctx.end();
186 if (it != end && (*it == 'a' || *it == 'p' || *it == 'n' || *it == 'd' || *it == 'o'))
187 presentation = *it++;
188
189 if (it != end && *it != '}')
190 throw format_error("invalid format");
191
192 return it;
193 }
194
195 template <typename FormatContext>
196 auto format(tattletale::Kernel &kernel, FormatContext &ctx) -> decltype(ctx.out())
197 {
198 if (presentation == 'a')
199 {
200 return fmt::formatter<std::string>::format(kernel.GetActiveDescription(), ctx);
201 }
202 else if (presentation == 'p')
203 {
204 return fmt::formatter<std::string>::format(kernel.GetPassiveDescription(), ctx);
205 }
206 else if (presentation == 'n')
207 {
208 return fmt::formatter<std::string>::format(kernel.name_, ctx);
209 }
210 else if (presentation == 'o')
211 {
212 return fmt::formatter<std::string>::format(kernel.GetDetailedDescription(), ctx);
213 }
214 return fmt::formatter<std::string>::format(kernel.GetDefaultDescription(), ctx);
215 }
