62template <typename T>
63struct fmt::formatter<T, std::enable_if_t<std::is_base_of<tattletale::Kernel, T>::value, char>> : fmt::formatter<std::string>
64{
65 // d - default, a - active, p - passive, n - name, o - detailed
66 char presentation = 'd';
67
68 constexpr auto parse(format_parse_context &ctx) -> decltype(ctx.begin())
69 {
70 auto it = ctx.begin(), end = ctx.end();
71 if (it != end && (*it == 'a' || *it == 'p' || *it == 'n' || *it == 'd' || *it == 'o'))
72 presentation = *it++;
73
74 if (it != end && *it != '}')
75 throw format_error("invalid format");
76
77 return it;
78 }
79
80 template <typename FormatContext>
81 auto format(tattletale::Kernel &kernel, FormatContext &ctx) -> decltype(ctx.out())
82 {
83 if (presentation == 'a')
84 {
85 return fmt::formatter<std::string>::format(kernel.GetActiveDescription(), ctx);
86 }
87 else if (presentation == 'p')
88 {
89 return fmt::formatter<std::string>::format(kernel.GetPassiveDescription(), ctx);
90 }
91 else if (presentation == 'n')
92 {
93 return fmt::formatter<std::string>::format(kernel.name_, ctx);
94 }
95 else if (presentation == 'o')
96 {
97 return fmt::formatter<std::string>::format(kernel.GetDetailedDescription(), ctx);
98 }
99 return fmt::formatter<std::string>::format(kernel.GetDefaultDescription(), ctx);
100 }
101};
