259{
260 // d - default = fullname, l - last name, s - second name, o - detail
261 char presentation = 'd';
262
263 constexpr auto parse(format_parse_context &ctx) -> decltype(ctx.begin())
264 {
265 auto it = ctx.begin(), end = ctx.end();
266 if (it != end && (*it == 'f' || *it == 'l' || *it == 'd' || *it == 'o'))
267 presentation = *it++;
268
269 if (it != end && *it != '}')
270 throw format_error("invalid format");
271
272 return it;
273 }
274
275 template <typename FormatContext>
276 auto format(tattletale::Actor &actor, FormatContext &ctx) -> decltype(ctx.out())
277 {
278 if (presentation == 'f')
279 {
280 return fmt::formatter<std::string>::format(actor.first_name_, ctx);
281 }
282 else if (presentation == 'l')
283 {
284 return fmt::formatter<std::string>::format(actor.last_name_, ctx);
285 }
286 else if (presentation == 'o')
287 {
288 return fmt::formatter<std::string>::format(actor.GetDetailedDescriptionString(), ctx);
289 }
290 return fmt::formatter<std::string>::format(actor.name_, ctx);
291 }
