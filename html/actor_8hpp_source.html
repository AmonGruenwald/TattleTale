257 // d - default = fullname, l - last name, s - second name, o - detail
258 char presentation = 'd';
259
260 constexpr auto parse(format_parse_context &ctx) -> decltype(ctx.begin())
261 {
262 auto it = ctx.begin(), end = ctx.end();
263 if (it != end && (*it == 'f' || *it == 'l' || *it == 'd' || *it == 'o'))
264 presentation = *it++;
265
266 if (it != end && *it != '}')
267 throw format_error("invalid format");
268
269 return it;
270 }
271
272 template <typename FormatContext>
273 auto format(tattletale::Actor &actor, FormatContext &ctx) -> decltype(ctx.out())
274 {
275 if (presentation == 'f')
276 {
277 return fmt::formatter<std::string>::format(actor.first_name_, ctx);
278 }
279 else if (presentation == 'l')
280 {
281 return fmt::formatter<std::string>::format(actor.last_name_, ctx);
282 }
283 else if (presentation == 'o')
284 {
285 return fmt::formatter<std::string>::format(actor.GetDetailedDescriptionString(), ctx);
286 }
287 return fmt::formatter<std::string>::format(actor.name_, ctx);
288 }
