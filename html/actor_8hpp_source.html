334 // d - default = fullname, l - last name, f - first name, o - detail
335 char presentation = 'd';
336
337 constexpr auto parse(format_parse_context &ctx) -> decltype(ctx.begin())
338 {
339 auto it = ctx.begin(), end = ctx.end();
340 if (it != end && (*it == 'f' || *it == 'l' || *it == 'd' || *it == 'o'))
341 presentation = *it++;
342
343 if (it != end && *it != '}')
344 throw format_error("invalid format");
345
346 return it;
347 }
348
349 template <typename FormatContext>
350 auto format(tattletale::Actor &actor, FormatContext &ctx) -> decltype(ctx.out())
351 {
352 if (presentation == 'd')
353 {
354 presentation = 'c';
355 if (actor.named_actors_)
356 {
357 if (actor.named_actors_->find(actor.id_) == actor.named_actors_->end())
358 {
359 actor.named_actors_->insert(actor.id_);
360 }
361 else
362 {
363 presentation = 'f';
364 }
365 }
366 }
367
368 if (presentation == 'f')
369 {
370 return fmt::formatter<std::string>::format(actor.first_name_, ctx);
371 }
372 else if (presentation == 'l')
373 {
374 return fmt::formatter<std::string>::format(actor.last_name_, ctx);
375 }
376 else if (presentation == 'o')
377 {
378 return fmt::formatter<std::string>::format(actor.GetDetailedDescriptionString(), ctx);
379 }
380 else if (presentation == 'c')
381 {
382 return fmt::formatter<std::string>::format(actor.name_, ctx);
383 }
384 }
