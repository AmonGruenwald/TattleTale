159
160 constexpr auto parse(format_parse_context &ctx) -> decltype(ctx.begin())
161 {
162 auto it = ctx.begin(), end = ctx.end();
163 if (it != end && (*it == 'd' || *it == 'a' || *it == 'p'))
164 presentation = *it++;
165
166 if (it != end && *it != '}')
167 throw format_error("invalid format");
168
169 return it;
170 }
171
172 template <typename FormatContext>
173 auto format(const tattletale::Verb &verb, FormatContext &ctx) -> decltype(ctx.out())
174 {
175 if (presentation == 'p')
176 {
177 return format_to(ctx.out(), "{}", verb.passive_verb);
178 }
179 else if (presentation == 'a')
180 {
181 return format_to(ctx.out(), "{}", verb.active_verb);
182 }
183
